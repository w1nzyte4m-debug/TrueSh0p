/* * TKK Forum — Неоновый имиджборд с авторизацией и профилем * Запуск: npm install ws formidable && node tkkforum.js */ const http = require('http'); const fs = require('fs'); const path = require('path'); const url = require('url'); const WebSocket = require('ws'); const formidable = require('formidable'); // ================== КОНФИГУРАЦИЯ ================== const PORT = process.env.PORT || 3000; const DATA_FILE = path.join(__dirname, 'data.json'); const UPLOAD_DIR = path.join(__dirname, 'uploads'); const MAX_THREADS = 50; const MAX_POSTS_PER_THREAD = 500; const THREAD_BUMP_LIMIT = 500; // ================== ХРАНИЛИЩЕ ================== let store = { users: {}, // username -> { password, token } threads: [], nextThreadId: 1, nextPostId: 1 }; try { if (fs.existsSync(DATA_FILE)) { const raw = JSON.parse(fs.readFileSync(DATA_FILE, 'utf8')); store.users = raw.users || {}; store.threads = raw.threads || []; store.nextThreadId = raw.nextThreadId || 1; store.nextPostId = raw.nextPostId || 1; } } catch (e) { console.error('Ошибка загрузки данных:', e); } if (!fs.existsSync(UPLOAD_DIR)) { fs.mkdirSync(UPLOAD_DIR, { recursive: true }); } function saveData() { try { fs.writeFileSync(DATA_FILE, JSON.stringify(store, null, 2), 'utf8'); } catch (e) { console.error('Ошибка сохранения:', e); } } // ================== УТИЛИТЫ ================== function genToken() { return Math.random().toString(36).substr(2) + Date.now().toString(36); } function getThreadById(id) { return store.threads.find(t => t.id === id); } function cleanOldThreads() { while (store.threads.length > MAX_THREADS) { const removed = store.threads.pop(); if (removed.image) fs.unlink(path.join(UPLOAD_DIR, removed.image), () => {}); removed.posts.forEach(p => { if (p.image) fs.unlink(path.join(UPLOAD_DIR, p.image), () => {}); }); } } function bumpThread(thread) { thread.bumpAt = Date.now(); store.threads = store.threads.filter(t => t.id !== thread.id); store.threads.unshift(thread); } // ================== WEBSOCKET ================== const wss = new WebSocket.Server({ noServer: true }); const clients = new Map(); // ws -> { username, token } function broadcast(msg) { const data = JSON.stringify(msg); wss.clients.forEach(client => { if (client.readyState === WebSocket.OPEN) { client.send(data); } }); } wss.on('connection', (ws) => { clients.set(ws, null); ws.on('message', (raw) => { let msg; try { msg = JSON.parse(raw); } catch { return; } // ==== Регистрация ==== if (msg.type === 'register') { const { username, password } = msg; if (!username || !password) { return ws.send(JSON.stringify({ type: 'regError', message: 'Заполните все поля' })); } if (store.users[username]) { return ws.send(JSON.stringify({ type: 'regError', message: 'Пользователь уже существует' })); } store.users[username] = { password, token: null }; saveData(); ws.send(JSON.stringify({ type: 'regSuccess' })); } // ==== Авторизация ==== else if (msg.type === 'login') { const { username, password } = msg; const user = store.users[username]; if (!user || user.password !== password) { return ws.send(JSON.stringify({ type: 'authError', message: 'Неверный логин или пароль' })); } const token = genToken(); user.token = token; clients.set(ws, { username, token }); saveData(); ws.send(JSON.stringify({ type: 'authSuccess', username, token })); } // ==== Восстановление сессии ==== else if (msg.type === 'restoreSession') { const { username, token } = msg; const user = store.users[username]; if (user && user.token === token) { clients.set(ws, { username, token }); ws.send(JSON.stringify({ type: 'authSuccess', username, token })); } else { ws.send(JSON.stringify({ type: 'authError', message: 'Сессия истекла' })); } } // ==== Создание треда ==== else if (msg.type === 'createThread') { const client = clients.get(ws); const isAuth = client && client.username && store.users[client.username]?.token === client.token; const authorName = isAuth ? client.username : (msg.name?.trim() || 'Аноним'); const title = msg.title?.trim() || 'Без названия'; const text = msg.text?.trim() || ''; const image = msg.image || null; // имя файла уже должно быть обработано const thread = { id: store.nextThreadId++, title, author: authorName, text, image, createdAt: Date.now(), bumpAt: Date.now(), posts: [] }; store.threads.unshift(thread); cleanOldThreads(); saveData(); broadcast({ type: 'threadCreated', threadPreview: { id: thread.id, title: thread.title, author: thread.author, text: thread.text.slice(0, 200), image: thread.image, postCount: 0, bumpAt: thread.bumpAt, createdAt: thread.createdAt } }); ws.send(JSON.stringify({ type: 'threadCreated', success: true, threadId: thread.id })); } // ==== Создание поста ==== else if (msg.type === 'createPost') { const client = clients.get(ws); const thread = getThreadById(msg.threadId); if (!thread) return; if (thread.posts.length >= MAX_POSTS_PER_THREAD) return; const isAuth = client && client.username && store.users[client.username]?.token === client.token; const authorName = isAuth ? client.username : (msg.name?.trim() || 'Аноним'); const text = msg.text?.trim() || ''; const image = msg.image || null; const post = { id: store.nextPostId++, author: authorName, text, image, createdAt: Date.now() }; thread.posts.push(post); if (thread.posts.length <= THREAD_BUMP_LIMIT) bumpThread(thread); cleanOldThreads(); saveData(); broadcast({ type: 'postAdded', threadId: thread.id, post: post }); ws.send(JSON.stringify({ type: 'postCreated', success: true, postId: post.id })); } // ==== Запрос списка тредов ==== else if (msg.type === 'getThreads') { ws.send(JSON.stringify({ type: 'threadsList', threads: store.threads.map(t => ({ id: t.id, title: t.title, author: t.author, text: t.text.slice(0, 200), image: t.image, postCount: t.posts.length, bumpAt: t.bumpAt, createdAt: t.createdAt })) })); } // ==== Запрос одного треда ==== else if (msg.type === 'getThread') { const thread = getThreadById(msg.threadId); if (thread) { ws.send(JSON.stringify({ type: 'threadDetail', thread })); } } // ==== Профиль пользователя ==== else if (msg.type === 'getProfile') { const username = msg.username; const threads = store.threads.filter(t => t.author === username || t.posts.some(p => p.author === username)); const profile = { username, threads: threads.map(t => ({ id: t.id, title: t.title, postCount: t.posts.length, bumpAt: t.bumpAt })), postsCount: store.threads.reduce((count, t) => { return count + t.posts.filter(p => p.author === username).length + (t.author === username ? 1 : 0); }, 0) }; ws.send(JSON.stringify({ type: 'profileData', profile })); } }); ws.on('close', () => { clients.delete(ws); }); }); // ================== HTTP СЕРВЕР ================== const server = http.createServer((req, res) => { const parsedUrl = url.parse(req.url, true); const pathname = parsedUrl.pathname; const method = req.method; if (method === 'GET') { if (pathname === '/' || pathname === '/index.html') { return serveIndex(res); } if (pathname.startsWith('/uploads/')) { return serveUpload(pathname, res); } if (pathname.startsWith('/api/profile/')) { const username = pathname.split('/api/profile/')[1]; const userExists = !!store.users[username]; res.writeHead(200, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ exists: userExists })); return; } } // API загрузки файлов (для треда и поста) if (method === 'POST') { if (pathname === '/api/upload') { return handleFileUpload(req, res); } } res.writeHead(404, { 'Content-Type': 'text/plain; charset=utf-8' }); res.end('Not Found'); }); function serveIndex(res) { res.writeHead(200, { 'Content-Type': 'text/html; charset=utf-8' }); res.end(getIndexHTML()); } function serveUpload(pathname, res) { const fileName = pathname.replace('/uploads/', ''); const filePath = path.join(UPLOAD_DIR, fileName); if (!fs.existsSync(filePath)) { res.writeHead(404); return res.end('File not found'); } const ext = path.extname(fileName).toLowerCase(); const mimeTypes = { '.jpg': 'image/jpeg', '.jpeg': 'image/jpeg', '.png': 'image/png', '.gif': 'image/gif', '.webp': 'image/webp' }; res.writeHead(200, { 'Content-Type': mimeTypes[ext] || 'application/octet-stream' }); fs.createReadStream(filePath).pipe(res); } function handleFileUpload(req, res) { const form = new formidable.IncomingForm({ uploadDir: UPLOAD_DIR, keepExtensions: true, maxFileSize: 10 * 1024 * 1024, filter: ({ mimetype }) => mimetype && mimetype.startsWith('image/') }); form.parse(req, (err, fields, files) => { if (err) { res.writeHead(500, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ error: 'Ошибка загрузки' })); return; } const imageFile = files.image; if (!imageFile || imageFile.size === 0) { return res.end(JSON.stringify({ image: null })); } const ext = path.extname(imageFile.originalFilename) || '.jpg'; const imageName = `${Date.now()}_${Math.random().toString(36).slice(2)}${ext}`; fs.renameSync(imageFile.filepath, path.join(UPLOAD_DIR, imageName)); res.writeHead(200, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ image: imageName })); }); } // ================== HTML ================== function getIndexHTML() { return ` TKK Forum – Авторизация и профиль

TKK Forum

Анонимный неоновый имиджборд с профилями

×
Вход
Регистрация
Новый тред
`; } // ================== ПОДКЛЮЧЕНИЕ WS К HTTP ================== server.on('upgrade', (request, socket, head) => { wss.handleUpgrade(request, socket, head, (ws) => { wss.emit('connection', ws, request); }); }); server.listen(PORT, () => console.log(`TKK Forum запущен на http://localhost:${PORT}`));
