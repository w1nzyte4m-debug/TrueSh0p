} } function generateId() { return Math.random().toString(36).substr(2, 9); } // ================== HTTP СЕРВЕР ================== const server = http.createServer((req, res) => { if (req.url === '/' || req.url === '/index.html') { res.writeHead(200, { 'Content-Type': 'text/html; const onlineUsers = new Map(); // username -> WebSocket function broadcastOnlineCount() { const count = onlineUsers.size; const msg = JSON.stringify({ type: 'onlineCount', count }); wss.clients.forEach(client => { if (client.readyState === WebSocket.OPEN) { client.send(msg); } }); } function sendToUser(username, messageObj) { const ws = onlineUsers.get(username); if (ws && ws.readyState === WebSocket.OPEN) { ws.send(JSON.stringify(messageObj)); } } function broadcastGlobalMessage(msg) { data.globalMessages.push(msg); if (data.globalMessages.length > 100) data.globalMessages.shift(); saveData(); const payload = JSON.stringify({ type: 'globalMessage', message: msg }); wss.clients.forEach(client => { if (client.readyState === WebSocket.OPEN) { client.send(payload); } }); } function getRoomId(user1, user2) { return [user1, user2].sort().join('::'); } function getOrCreatePrivateRoom(user1, user2) { const roomId = getRoomId(user1, user2); if (!data.privateMessages[roomId]) { data.privateMessages[roomId] = []; } return roomId; } function sendPrivateMessage(roomId, from, text) { const room = data.privateMessages[roomId]; if (!room) return; const msg = { from, text, time: new Date().toISOString() }; room.push(msg); saveData(); const payload = JSON.stringify({ type: 'privateMessage', roomId, message: msg }); const users = roomId.split('::'); users.forEach(username => { sendToUser(username, { ...JSON.parse(payload), type: 'privateMessage' }); }); } wss.on('connection', (ws) => { let currentUser = null; ws.on('message', (raw) => { let msg; try { msg = JSON.parse(raw); } catch (e) { return; } // Авторизация if (msg.type === 'auth') { const { username, password } = msg; const user = data.users[username]; if (user && user.password === password) { user.token = generateId(); currentUser = username; onlineUsers.set(username, ws); broadcastOnlineCount(); ws.send(JSON.stringify({ type: 'authSuccess', username, token: user.token, products: data.products, globalMessages: data.globalMessages.slice(-50), onlineCount: onlineUsers.size })); } else { ws.send(JSON.stringify({ type: 'authError', message: 'Неверный логин или пароль' })); } } // Регистрация (исправленная — автоматический вход) else if (msg.type === 'register') { const { username, password } = msg; if (!username || !password) { ws.send(JSON.stringify({ type: 'regError', message: 'Заполните все поля' })); } else if (data.users[username]) { ws.send(JSON.stringify({ type: 'regError', message: 'Пользователь уже существует' })); } else { // Создаём пользователя data.users[username] = { password, token: null }; saveData(); // Сразу генерируем токен и авторизуем const newUser = data.users[username]; newUser.token = generateId(); currentUser = username; onlineUsers.set(username, ws); broadcastOnlineCount(); ws.send(JSON.stringify({ type: 'authSuccess', username, token: newUser.token, products: data.products, globalMessages: data.globalMessages.slice(-50), onlineCount: onlineUsers.size })); } } // Добавление товара else if (msg.type === 'addProduct') { if (!currentUser || data.users[currentUser]?.token !== msg.token) { ws.send(JSON.stringify({ type: 'error', message: 'Не авторизован' })); return; } const { name, description, price, category } = msg; if (!name || !description || isNaN(price) || !category) { ws.send(JSON.stringify({ type: 'addProductError', message: 'Некорректные данные' })); return; } const product = { id: generateId(), name, description, price: parseFloat(price), category, seller: currentUser, createdAt: new Date().toISOString() }; data.products.push(product); saveData(); const broadcast = JSON.stringify({ type: 'newProduct', product }); wss.clients.forEach(client => { if (client.readyState === WebSocket.OPEN) client.send(broadcast); }); } // Глобальное сообщение else if (msg.type === 'globalMessage') { if (!currentUser || data.users[currentUser]?.token !== msg.token) return; broadcastGlobalMessage({ from: currentUser, text: msg.text, time: new Date().toISOString() }); } // Начать приватный чат else if (msg.type === 'startPrivateChat') { if (!currentUser || data.users[currentUser]?.token !== msg.token) return; const { withUser } = msg; const roomId = getOrCreatePrivateRoom(currentUser, withUser); const history = data.privateMessages[roomId] || []; ws.send(JSON.stringify({ type: 'privateChatHistory', roomId, withUser, messages: history })); sendToUser(withUser, { type: 'privateChatOpened', roomId, withUser: currentUser }); } // Приватное сообщение else if (msg.type === 'privateMessage') { if (!currentUser || data.users[currentUser]?.token !== msg.token) return; const { roomId, text } = msg; sendPrivateMessage(roomId, currentUser, text); } }); ws.on('close', () => { if (currentUser) { onlineUsers.delete(currentUser); broadcastOnlineCount(); } }); }); // ================== HTML-СТРАНИЦА ================== function getHTML() { return ` TrueShop | Неоновый маркет
