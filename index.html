` + `Ammo: ${w.ammo}/${w.maxAmmo}`; } /* ---------- NPC ---------- */ const npcs = []; function createNPC(x, z) { const npc = new THREE.Mesh( new THREE.BoxGeometry(1, 2, 1), new THREE.MeshStandardMaterial({ color: 0xff4444 }) ); npc.position.set(x, 1, z); npc.userData = { hp: 100 }; scene.add(npc); npcs.push(npc); } /* ---------- СПАВН NPC ---------- */ for(let i = 0; i < 15; i++) { createNPC( (Math.random() - 0.5) * 100, (Math.random() - 0.5) * 100 ); } /* ---------- ВЫБОР ОРУЖИЯ ---------- */ window.addEventListener("keydown", e => { if(e.code === "Digit1") currentWeapon = "rifle"; if(e.code === "Digit2") currentWeapon = "pistol"; }); /* ---------- СТРЕЛЬБА ---------- */ const raycaster = new THREE.Raycaster(); window.addEventListener("mousedown", () => { const weapon = weapons[currentWeapon]; if(Date.now() - lastShot < weapon.fireRate) return; if(weapon.ammo <= 0) return; lastShot = Date.now(); weapon.ammo--; raycaster.setFromCamera( new THREE.Vector2(0,0), camera ); const hits = raycaster.intersectObjects(npcs); if(hits.length > 0) { const npc = hits[0].object; npc.userData.hp -= weapon.damage; if(npc.userData.hp <= 0) { scene.remove(npc); const index = npcs.indexOf(npc); if(index !== -1) npcs.splice(index,1); } } updateHUD(); }); /* ---------- ПЕРЕЗАРЯДКА ---------- */ window.addEventListener("keydown", e => { if(e.code === "KeyR") { const w = weapons[currentWeapon]; w.ammo = w.maxAmmo; updateHUD(); } }); /* ---------- ПРОСТОЙ ИИ ---------- */ function updateNPCs() { npcs.forEach(npc => { const dir = new THREE.Vector3( camera.position.x - npc.position.x, 0, camera.position.z - npc.position.z ); const dist = dir.length(); if(dist < 30) { dir.normalize(); npc.position.x += dir.x * 0.02; npc.position.z += dir.z * 0.02; } }); } /* ---------- В ОСНОВНОЙ ЦИКЛ ---------- */ // добавить внутрь animate(): updateNPCs(); updateHUD();
