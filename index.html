/* * TKK Forum — анонимный неоновый имиджборд * Запуск: npm install ws formidable && node tkkforum.js */ const http = require('http'); const fs = require('fs'); const path = require('path'); const url = require('url'); const WebSocket = require('ws'); const formidable = require('formidable'); // ================== КОНФИГУРАЦИЯ ================== const PORT = process.env.PORT || 3000; const DATA_FILE = path.join(__dirname, 'data.json'); const UPLOAD_DIR = path.join(__dirname, 'uploads'); const MAX_THREADS = 50; const MAX_POSTS_PER_THREAD = 500; const THREAD_BUMP_LIMIT = 500; // бамп, если меньше этого числа постов // ================== ХРАНИЛИЩЕ ================== let store = { threads: [], nextThreadId: 1, nextPostId: 1 }; try { if (fs.existsSync(DATA_FILE)) { store = JSON.parse(fs.readFileSync(DATA_FILE, 'utf8')); } } catch (e) { console.error('Ошибка загрузки данных:', e); } // Создаём папку для загрузок if (!fs.existsSync(UPLOAD_DIR)) { fs.mkdirSync(UPLOAD_DIR, { recursive: true }); } function saveData() { try { fs.writeFileSync(DATA_FILE, JSON.stringify(store, null, 2), 'utf8'); } catch (e) { console.error('Ошибка сохранения:', e); } } // ================== ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ ================== function generateThreadId() { return store.nextThreadId++; } function generatePostId() { return store.nextPostId++; } function getThreadById(id) { return store.threads.find(t => t.id === id); } function cleanOldThreads() { while (store.threads.length > MAX_THREADS) { const removed = store.threads.pop(); // удаляем самый старый (последний в массиве, т.к. новые в начало) // удаляем файлы постов из треда if (removed.image) { const imgPath = path.join(UPLOAD_DIR, removed.image); fs.unlink(imgPath, () => {}); } removed.posts.forEach(p => { if (p.image) { const imgPath = path.join(UPLOAD_DIR, p.image); fs.unlink(imgPath, () => {}); } }); } } function bumpThread(thread) { const now = Date.now(); thread.bumpAt = now; // Переместить тред в начало массива store.threads = store.threads.filter(t => t.id !== thread.id); store.threads.unshift(thread); } // ================== WEB SOCKET ================== const wss = new WebSocket.Server({ noServer: true }); const clients = new Set(); wss.on('connection', (ws) => { clients.add(ws); // При подключении отправляем актуальный список тредов (без постов, только превью) ws.send(JSON.stringify({ type: 'threadsList', threads: store.threads.map(t => ({ id: t.id, title: t.title, name: t.name || 'Аноним', text: t.text.slice(0, 200), // превью image: t.image, postCount: t.posts.length, bumpAt: t.bumpAt, createdAt: t.createdAt })) })); ws.on('close', () => clients.delete(ws)); }); function broadcast(msg) { const data = JSON.stringify(msg); clients.forEach(client => { if (client.readyState === WebSocket.OPEN) { client.send(data); } }); } // ================== HTTP СЕРВЕР ================== const server = http.createServer((req, res) => { const parsedUrl = url.parse(req.url, true); const pathname = parsedUrl.pathname; const method = req.method; // Раздача статики if (method === 'GET') { if (pathname === '/' || pathname === '/index.html') { return serveIndex(res); } if (pathname.startsWith('/uploads/')) { return serveUpload(pathname, res); } if (pathname === '/api/threads') { return apiGetThreads(res); } if (pathname.startsWith('/api/threads/') && pathname.endsWith('/posts')) { const threadId = parseInt(pathname.split('/')[2]); return apiGetThread(threadId, res); } } // API: создание треда if (method === 'POST' && pathname === '/api/threads') { return handleCreateThread(req, res); } // API: создание поста if (method === 'POST' && pathname.match(/^\/api\/threads\/(\d+)\/posts$/)) { const threadId = parseInt(pathname.split('/')[2]); return handleCreatePost(threadId, req, res); } // Всё остальное – 404 res.writeHead(404, { 'Content-Type': 'text/plain; charset=utf-8' }); res.end('Not Found'); }); // ================== ОБРАБОТКА ЗАПРОСОВ ================== function serveIndex(res) { const html = getIndexHTML(); res.writeHead(200, { 'Content-Type': 'text/html; charset=utf-8' }); res.end(html); } function serveUpload(pathname, res) { const fileName = pathname.replace('/uploads/', ''); const filePath = path.join(UPLOAD_DIR, fileName); if (!fs.existsSync(filePath)) { res.writeHead(404); return res.end('File not found'); } const ext = path.extname(fileName).toLowerCase(); const mimeTypes = { '.jpg': 'image/jpeg', '.jpeg': 'image/jpeg', '.png': 'image/png', '.gif': 'image/gif', '.webp': 'image/webp' }; res.writeHead(200, { 'Content-Type': mimeTypes[ext] || 'application/octet-stream' }); fs.createReadStream(filePath).pipe(res); } function apiGetThreads(res) { const list = store.threads.map(t => ({ id: t.id, title: t.title, name: t.name || 'Аноним', text: t.text.slice(0, 200), image: t.image, postCount: t.posts.length, bumpAt: t.bumpAt, createdAt: t.createdAt })); res.writeHead(200, { 'Content-Type': 'application/json' }); res.end(JSON.stringify(list)); } function apiGetThread(threadId, res) { const thread = getThreadById(threadId); if (!thread) { res.writeHead(404, { 'Content-Type': 'application/json' }); return res.end(JSON.stringify({ error: 'Thread not found' })); } res.writeHead(200, { 'Content-Type': 'application/json' }); res.end(JSON.stringify(thread)); } // Загрузка формы с файлом function parseForm(req) { return new Promise((resolve, reject) => { const form = new formidable.IncomingForm({ uploadDir: UPLOAD_DIR, keepExtensions: true, maxFileSize: 10 * 1024 * 1024, // 10 MB filter: ({ name, originalFilename, mimetype }) => { return mimetype && mimetype.startsWith('image/'); } }); form.parse(req, (err, fields, files) => { if (err) return reject(err); resolve({ fields, files }); }); }); } async function handleCreateThread(req, res) { try { const { fields, files } = await parseForm(req); const title = fields.title && fields.title.trim() || 'Без названия'; const name = fields.name && fields.name.trim() || ''; const text = fields.text && fields.text.trim() || ''; const imageFile = files.image; let imageName = null; if (imageFile && imageFile.size > 0) { // Переименовываем в уникальное имя const ext = path.extname(imageFile.originalFilename) || '.jpg'; imageName = `${Date.now()}_${Math.random().toString(36).slice(2)}${ext}`; fs.renameSync(imageFile.filepath, path.join(UPLOAD_DIR, imageName)); } const thread = { id: generateThreadId(), title, name: name || 'Аноним', text, image: imageName, createdAt: Date.now(), bumpAt: Date.now(), posts: [] }; store.threads.unshift(thread); cleanOldThreads(); saveData(); // Уведомление через WebSocket broadcast({ type: 'threadCreated', threadPreview: { id: thread.id, title: thread.title, name: thread.name, text: thread.text.slice(0, 200), image: thread.image, postCount: 0, bumpAt: thread.bumpAt, createdAt: thread.createdAt } }); res.writeHead(200, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ success: true, threadId: thread.id })); } catch (err) { console.error(err); res.writeHead(500, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ error: 'Ошибка сервера' })); } } async function handleCreatePost(threadId, req, res) { try { const thread = getThreadById(threadId); if (!thread) { res.writeHead(404, { 'Content-Type': 'application/json' }); return res.end(JSON.stringify({ error: 'Тред не найден' })); } if (thread.posts.length >= MAX_POSTS_PER_THREAD) { res.writeHead(403, { 'Content-Type': 'application/json' }); return res.end(JSON.stringify({ error: 'Достигнут лимит постов' })); } const { fields, files } = await parseForm(req); const name = fields.name && fields.name.trim() || ''; const text = fields.text && fields.text.trim() || ''; const imageFile = files.image; let imageName = null; if (imageFile && imageFile.size > 0) { const ext = path.extname(imageFile.originalFilename) || '.jpg'; imageName = `${Date.now()}_${Math.random().toString(36).slice(2)}${ext}`; fs.renameSync(imageFile.filepath, path.join(UPLOAD_DIR, imageName)); } const post = { id: generatePostId(), name: name || 'Аноним', text, image: imageName, createdAt: Date.now() }; thread.posts.push(post); // Бамп, если число постов меньше лимита if (thread.posts.length <= THREAD_BUMP_LIMIT) { bumpThread(thread); } cleanOldThreads(); saveData(); // Уведомление всем broadcast({ type: 'postAdded', threadId: thread.id, post: post }); res.writeHead(200, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ success: true, postId: post.id })); } catch (err) { console.error(err); res.writeHead(500, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ error: 'Ошибка сервера' })); } } // ================== HTML ГЛАВНОЙ СТРАНИЦЫ ================== function getIndexHTML() { return ` TKK Forum – Неоновый имиджборд

TKK Forum

Анонимный неоновый имиджборд • Добро пожаловать

Новый тред
×
`; } // ================== ПОДКЛЮЧАЕМ WS К СЕРВЕРУ ================== server.on('upgrade', (request, socket, head) => { wss.handleUpgrade(request, socket, head, (ws) => { wss.emit('connection', ws, request); }); }); // ================== ЗАПУСК ================== server.listen(PORT, () => { console.log(`TKK Forum запущен на http://localhost:${PORT}`); });
